lotr_generic_capital_mission = {
	header = "lotr_generic_mission_landscape_headers"
	icon = "lotr_generic_capital_mission_icon"

	repeatable = no

	chance = {}

	ai_chance = {
		factor = 0
	}

	potential = {
    	NOT = { has_variable = lotr_generic_mission_activated_variable }
		has_civil_war = no
	}

	on_potential = {
		random_owned_province = {
			limit = {
                owner = root
                is_capital = yes
			} 
			save_scope_as = lotr_actual_capital
		}
	}

	abort = {
		ai_mission_back_out_trigger = yes
	}

	on_start = {
		if = {
			limit = {
				root = { is_tribal = yes }
			}
			capital_scope = {
				create_state_pop = tribesmen
				create_state_pop = tribesmen
			}
			add_stability = 5
		}
		else = {
			add_stability = 5
		}
		start_mission_ai_effect = yes
	}

	on_abort = {
		add_stability = -15
        set_variable = {
            name = lotr_generic_mission_activated_variable
        }
	}

	on_completion = {
		complete_mission_effect = yes
	}
	
############################# Province missions #####################################

	lotr_generic_capital_mission_1 = {		# Start
		icon = "task_economical"

    	duration = 220

		allow = {
			treasury >= 80
			political_influence >= 20
			war = no
		}
		
		highlight = {
			scope:province = { this = root.capital_scope }
		}

		on_start = {
            add_treasury = -80
			add_political_influence = -20
		}

		on_completion = {
			current_ruler = {
				add_popularity = popularity_medium
			}
			add_1_free_province_investments = yes
			if = {
				limit = {
					root = { is_tribal = yes }
				}
				add_centralization = 5
				add_stability = 10
			}
			else = {
				add_stability = 5
			}
		} 
	}
