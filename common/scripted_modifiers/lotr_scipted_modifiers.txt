	modifier = {
		add = {
			value = population_cap
			subtract = total_population
			multiply = 2

		}

	modifier = {
		factor = 4
		NOT = {
			any_neighbor_province = {
				is_inhabitable = yes
				OR = {
					has_owner = no
					owner = { country_culture_group = orcs_group }
				}
			}
		}
	}
}
