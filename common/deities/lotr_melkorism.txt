		polytheistic_check_religion_trigger = yes

			common_deity_trigger = {
				RELIGION = melkorism
			}
			deity:omen_sauron = {
				holy_site_deity_check_trigger = yes
			}
		}
	}
	icon = deity_sauron

	religion = melkorism
	deity_category = war #This Deity uses a War slot.

	deification_trigger = {
		can_deify_trigger = yes
	}
	
	on_activate = {
		military_apotheosis_manpower_effect = yes
	}
}
