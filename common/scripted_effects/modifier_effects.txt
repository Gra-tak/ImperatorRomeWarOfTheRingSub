
add_character_modifier_stack = {
	if = {
		limit = {
			has_character_modifier = $MOD$
		}
		remove_character_modifier = $MOD$
	}
	while = {
		count = $NUM$
		add_character_modifier = {
			name = $MOD$
			mode = add
			duration = $DUR$
		}
	}
}
